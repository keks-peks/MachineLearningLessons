    year = int(d[5:9])
    month = int(d[9:11])
    day = int(d[11:13])
    hour = int(d[13:15])
    # Humidity
    rh = int(d[15:18])
    # Height of the base of the lowest cloud
    hcld = int(d[43:45])
    # Dew point
    tdew = int(d[47:50])
    # Cloud amount
    tcld = int(d[53:55])
    # Wind direction
    wdir = int(d[61:63])
    # Wind speed
    wspd = int(d[64:66])
    # Air pressure
    stap = float(d[73:78])/10
    # Past weather code
    w = int(d[83:84])
    # Air temperature
    airt = float(d[89:93]) / 10 if d[89:93] != '9999' else 0.0
    if hour:
        meteo_x.append(airt)       
        meteo_y.append(airt)
